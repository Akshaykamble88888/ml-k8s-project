name: Auto Deploy to AWS

on:
  push:
    branches:
      - main

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout Code
      uses: actions/checkout@v2

    - name: Deploy to EC2
      uses: appleboy/ssh-action@master
      with:
        host: ${{ secrets.EC2_HOST }}
        username: ubuntu
        key: ${{ secrets.EC2_KEY }}
        script: |
          # Clone repo if not exists
          if [ ! -d "ml-project" ]; then
            git clone https://github.com/YOUR_USERNAME/YOUR_REPO.git ml-project
          fi

          cd ml-project
          git pull origin main

          # Stop old container
          docker stop ml-app || true
          docker rm ml-app || true

          # Build new image
          docker build -t ml-app .

          # Run new container
          docker run -d -p 5000:5000 --name ml-app ml-app
